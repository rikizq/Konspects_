import requests
from aiogram import Bot, Dispatcher, types
from aiogram.utils import executor
import random

TOKEN = "8205786674:AAF0JYnQBU7F6-hXQ0eqjYoJZyxAFhlxKsA"

bot = Bot(token=TOKEN)
dp = Dispatcher(bot)

# --- Функции поиска ---
def search_music(query, limit=5):
    url = "https://itunes.apple.com/search"
    params = {"term": query, "entity": "song", "limit": limit}
    return requests.get(url, params=params).json().get("results", [])

def search_by_artist(artist, limit=5):
    url = "https://itunes.apple.com/search"
    params = {"term": artist, "entity": "song", "attribute": "artistTerm", "limit": limit}
    return requests.get(url, params=params).json().get("results", [])

def search_by_track(track, limit=5):
    url = "https://itunes.apple.com/search"
    params = {"term": track, "entity": "song", "attribute": "songTerm", "limit": limit}
    return requests.get(url, params=params).json().get("results", [])

def get_top_tracks(limit=5):
    url = f"https://itunes.apple.com/us/rss/topsongs/limit={limit}/json"
    results = requests.get(url).json().get("feed", {}).get("entry", [])
    tracks = []
    for t in results:
        tracks.append({
            "trackName": t["im:name"]["label"],
            "artistName": t["im:artist"]["label"],
            "previewUrl": t["link"][0]["attributes"]["href"]
        })
    return tracks

def get_random_tracks(limit=5):
    results = search_music("a", limit=50)
    return random.sample(results, min(limit, len(results)))

# --- Команды бота ---
@dp.message_handler(commands=['start'])
async def cmd_start(message: types.Message):
    await message.answer(
        "Привет! Я музыкальный бот 🎵\n\n"
        "Напиши название трека или исполнителя, и я найду музыку для тебя.\n"
        "Используй /help чтобы узнать все команды."
    )

@dp.message_handler(commands=['help'])
async def cmd_help(message: types.Message):
    await message.answer(
        "Вот что я умею:\n"
        "/help - показать эту справку\n"
        "/random - получить случайные треки\n"
        "/top - показать топ-песни\n"
        "/settings - настройки бота (заглушка)\n"
        "/artist <имя> - поиск треков по исполнителю\n"
        "/track <название> - поиск конкретного трека\n\n"
        "Или просто напиши название трека или исполнителя."
    )

@dp.message_handler(commands=['random'])
async def cmd_random(message: types.Message):
    tracks = get_random_tracks()
    text = " Случайные треки:\n\n"
    for t in tracks:
        text += f"🎵 {t['trackName']}\n👤 {t['artistName']}\n{t['previewUrl']}\n\n"
    await message.answer(text)

@dp.message_handler(commands=['top'])
async def cmd_top(message: types.Message):
    tracks = get_top_tracks()
    text = " Топ-песни:\n\n"

        text += f"🎵 {t['trackName']}\n👤 {t['artistName']}\n{t['previewUrl']}\n\n"
    await message.answer(text)

    await message.answer(" Настройки бота пока недоступны, но скоро будут!")

# --- Поиск по исполнителю ---
@dp.message_handler(commands=['artist'])
async def cmd_artist(message: types.Message):
    args = message.get_args()
    if not args:
        await message.answer(" Пожалуйста, укажи имя исполнителя после команды /artist")
        return
    tracks = search_by_artist(args)

        await message.answer("Ничего не найдено ")
        return

    text = f"🎤 Треки исполнителя {args}:\n\n"

        text += f"🎵 {t['trackName']}\n👤 {t['artistName']}\n{t['previewUrl']}\n\n"
    await message.answer(text)

# --- Поиск конкретного трека ---
@dp.message_handler(commands=['track'])
async def cmd_track(message: types.Message):
    args = message.get_args()
    if not args:
        await message.answer(" Пожалуйста, укажи название трека после команды /track")
        return
    tracks = search_by_track(args)
    if not tracks:
        await message.answer("Ничего не найдено ")
        return

    text = f"🎵 Результаты поиска трека '{args}':\n\n"
    for t in tracks:
        text += f"🎵 {t['trackName']}\n👤 {t['artistName']}\n{t['previewUrl']}\n\n"
    await message.answer(text)

# --- Любой текст (поиск по общему запросу) ---
@dp.message_handler()
async def music_search(message: types.Message):

    text = "🎶 Результаты:\n\n"

        text += f"🎵 {t['trackName']}\n {t['artistName']}\n{t['previewUrl']}\n\n"
    await message.answer(text)

# --- Запуск бота ---
