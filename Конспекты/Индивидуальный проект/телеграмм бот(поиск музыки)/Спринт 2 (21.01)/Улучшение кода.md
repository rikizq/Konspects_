from aiogram.filters import Command
from aiogram.types import Message

def search_music(query, limit=SEARCH_LIMIT):
    url = "https://itunes.apple.com/search"
    params = {
        "term": query,
        "entity": "song",
        "limit": limit,
        "country": "RU",
        "lang": "ru_ru"
    }
    return requests.get(url, params=params).json().get("results", [])


def search_by_artist(artist, limit=SEARCH_LIMIT):
    return search_music(artist, limit)


def search_by_track(track, limit=SEARCH_LIMIT):
    return search_music(track, limit)


def get_top_tracks(limit=SEARCH_LIMIT):
    url = f"https://itunes.apple.com/ru/rss/topsongs/limit={limit}/json"
    try:
        data = requests.get(url).json()
        entries = data["feed"]["entry"]
        tracks = []
        for e in entries:
            tracks.append({
                "trackName": e["im:name"]["label"],
                "artistName": e["im:artist"]["label"],
                "previewUrl": e["link"][0]["attributes"]["href"]
            })
        return tracks
    except Exception:
        return []


def get_random_tracks(limit=SEARCH_LIMIT):
    terms = ["love", "pop", "rock", "a", "и", "me"]
    results = search_music(random.choice(terms), limit=20)
    return random.sample(results, min(limit, len(results)))

Мой код реализует Telegram-бота для поиска и прослушивания 30-секундных фрагментов треков через iTunes Search API от Apple Inc. Бот написан на библиотеке aiogram и работает в асинхронном режиме через polling.

В начале подключаются необходимые библиотеки aiogram для работы с Telegram Bot API, requests для отправки HTTP-запросов к музыкальному API, asyncio для асинхронного запуска и random для генерации случайных треков. Затем создаются объекты Bot и Dispatcher, которые отвечают за обработку сообщений и команд.

Также реализован блок функций для работы с iTunes API. Функция search_music() отправляет GET-запрос на https://itunes.apple.com/search и возвращает список найденных песен в формате JSON. Функции search_by_artist() и search_by_track() используют тот же механизм, но применяются для более конкретного поиска. get_top_tracks() получает топ-песни через RSS-ленту iTunes, а get_random_tracks() выбирает случайные результаты из общего поиска.

Функция send_preview() отвечает за отправку аудио пользователю. Она берёт из ответа API поле previewUrl - это официальный 30-секундный фрагмент из каталога iTunes Store / Apple Music - и передаёт его в метод answer_audio, чтобы Telegram отобразил встроенный плеер.
