```python

class Solution:
    def isPalindrome(self, x: int) -> bool:
        str_x = str(x)
        first = 0
        last = -1
        
        while True:
            print(first, last)
            if str_x[first] == str_x[last]:
                first += 1
                last -= 1
            else:
                return False

            if first + last == -1:
                print(first, last)
                return True
```
