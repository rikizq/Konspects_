function plusOne(digits: number[]): number[] {
    // Проходим по цифрам справа налево
    for (let i = digits.length - 1; i >= 0; i--) {
        // Если цифра меньше 9, просто увеличиваем на 1 и возвращаем результат
        if (digits[i] < 9) {
            digits[i]++;
            return digits;
        }
