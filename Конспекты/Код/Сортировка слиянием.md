```python
import random
import time


def merge_sort(A):
    if len(A) <= 1:
        return A

    mid = len(A) // 2
    left = merge_sort(A[:mid])
    right = merge_sort(A[mid:])

    return merge(left, right)


def merge(left, right):
    result = []
    i = j = 0

    while i < len(left) and j < len(right):
        if left[i] < right[j]:
            result.append(left[i])
            i += 1
        else:
            result.append(right[j])
            j += 1

    result.extend(left[i:])
    result.extend(right[j:])

    return result


def measure():
    sizes = [0, 1000, 2000, 3000, 4000, 5000]

    arrays = [
        [] if size == 0 else [random.randint(0, 100) for _ in range(size)]
        for size in sizes
    ]

    print('Прогрев: начат')
    for _ in range(50):
        for arr in arrays[1:4]:
            merge_sort(arr)
    print('Прогрев: окончен')

    for arr in arrays:
        average_time = 0
        for _ in range(10):
            start_time = time.perf_counter()
            merge_sort(arr)
            end_time = time.perf_counter()
            average_time += (end_time - start_time)

        avg = average_time / 10

        print(
            'Закончен подсчёт сортировки последовательности длиной:',
            len(arr),
            f"\t{avg:.6f} сек."
        )


if __name__ == "__main__":
    measure()
```
