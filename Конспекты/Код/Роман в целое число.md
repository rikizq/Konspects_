```python
class Solution:

    def romanToInt(self, s: str) -> int:
        romanA = {"I": 1, "V": 5, "X": 10, "L": 50, "C": 100, "D": 500, "M": 1000}
        num10 = 0
        prev_roman = " "
        
        for c in s[::-1]:
            prev_roman = c
            if prev_roman == "I" and c in ["V", "X"]:
                num10 += romanA.get(c) - 2
                print(romanA.get(c), romanA.get(c) - 1, "I")
                
            if prev_roman == "X" and c in ["C", "L"]:
                num10 += romanA.get(c) - 20
                print(romanA.get(c), romanA.get(c) - 10, "X")
                
            elif prev_roman == "C" and c in ["D", "M"]:
                num10 += romanA.get(c) - 200
                print(c, romanA.get(c), romanA.get(c) - 100, "C")
            else:
                num10 += romanA.get(c)
                print(romanA.get(c), "else")
            
            prev_roman + c
            
        return num10
```
