class Solution:
    def pivotInteger(self, n: int) -> int:
         # Используем формулу для суммы первых n чисел
        total_sum = n * (n + 1) // 2
        
        # Проверяем, является ли общая сумма полным квадратом
        root = int(total_sum ** 0.5)
        
        # Если возводимое в квадрат число даёт ровно нашу сумму, 
        # значит, это и есть опорное число
        if root * root == total_sum:
            return root
        else:
            return -1
