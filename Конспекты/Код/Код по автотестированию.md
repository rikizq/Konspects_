```python
import pytest
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.chrome.options import Options
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC

chrome_options = Options()
browser = webdriver.Chrome(options=chrome_options)
SITE_ADDRESS = 'https://practice-automation.com/'
CLICK_EVENTS_PAGE_ADDRESS = 'https://practice-automation.com/click-events/'
CLICK_EVENTS_BUTTON_CSS = 'a.wp-block-button__link.wp-element-button[href="https://practice-automation.com/click-events/"]'

CAT_BUTTON_CSS = 'a'

@pytest.fixture(scope='module')
def browser():
    chrome_options = Options()
    driver = webdriver.Chrome(options=chrome_options)
    yield driver
    driver.quit()

def test_openMainPage(browser):
    wait = WebDriverWait(browser, 10)
    browser.get(SITE_ADDRESS)
    assert browser.current_url == SITE_ADDRESS

def test_findAndClickEventsButton(browser):
    button = browser.find_element(By.CSS_SELECTOR, CLICK_EVENTS_BUTTON_CSS)
    browser.execute_script("arguments[0].click();", button)
    wait = WebDriverWait(browser, 10)
    assert browser.current_url == CLICK_EVENTS_PAGE_ADDRESS

def test_animalButtons(browser):
    cat_button = browser.find_element(By.XPATH, '//button[@onclick="catSound()"]')
    pig_button = browser.find_element(By.XPATH, '//button[@onclick="pigSound()"]')
    cow_button = browser.find_element(By.XPATH, '//button[@onclick="cowSound()"]')
    dog_button = browser.find_element(By.XPATH, '//button[@onclick="dogSound()"]')

    cat_button.click()
    demo_element = browser.find_element(By.ID, 'demo')
    assert demo_element.text == 'Meow!'

    pig_button.click()
    demo_element = browser.find_element(By.ID, 'demo')
    assert demo_element.text == 'Oink!'

    dog_button.click()
    demo_element = browser.find_element(By.ID, 'demo')
    assert demo_element.text == 'Woof!'

    cow_button.click()
    demo_element = browser.find_element(By.ID, 'demo')
    assert demo_element.text == 'Moo!'
```
