```python
import random
import time

def insertion_sort(A):
    A_copy = A.copy()
    for i in range(1, len(A_copy)):
        key = A_copy[i]
        j = i - 1
        while j >= 0 and A_copy[j] > key:
            A_copy[j + 1] = A_copy[j]
            j -= 1
        A_copy[j + 1] = key
    return A_copy


def measure():
    sizes = [0, 1000, 2000, 3000, 4000, 5000]

    arrays = [
        [] if size == 0 else [random.randint(0, 100) for _ in range(size)]
        for size in sizes
    ]

    print('Прогрев начат:')
    for _ in range(20):
        insertion_sort(arrays[1])
    print('Прогрев окончен:\n')

    times = []

    for arr in arrays:
        total = 0
        for _ in range(10):
            start = time.perf_counter()
            insertion_sort(arr)
            end = time.perf_counter()
            total += (end - start)

        avg = total / 10
        times.append(avg)

        print(f'Закончен подсчёт сортировки последовательности длиной: {len(arr):5d}     {avg:.6f} сек.')

    return sizes, times


if __name__ == "__main__":
    measure()
```
