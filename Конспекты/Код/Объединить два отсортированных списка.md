class Solution:
    def mergeTwoLists(self, list1: Optional[ListNode], list2: Optional[ListNode]) -> Optional[ListNode]:
        yarik = ListNode()
        masuka = yarik
        t = None
        while list1 and list2:
            if list1.val <= list2.val:
                yarik.val = list1.val
                list1 = list1.next
            else:    
                yarik.val = list2.val
                list2 = list2.next
            yarik.next = ListNode()
            t = yarik
            yarik = yarik.next
        while list1:
            yarik.val = list1.val
            list1 = list1.next
            yarik.next = ListNode()
            t = yarik
            yarik = yarik.next
        while list2:
            yarik.val = list2.val
            list2 = list2.next
            yarik.next = ListNode()
            t = yarik
            yarik = yarik.next
        if t:
            t.next = None    
        else:
            masuka = None
        return masuka
